# Load the dataset
data("ny_noaa")

# Clean and transform the data
ny_noaa_clean = ny_noaa %>%
  # Convert specified columns to character
  mutate(across(c(tmax, tmin, prcp, snow, snwd), as.character)) %>%
  # Replace "M" and "T" with NA
  mutate(across(c(tmax, tmin, prcp, snow, snwd), ~ na_if(., "M"))) %>%
  mutate(across(c(tmax, tmin, prcp, snow, snwd), ~ na_if(., "T"))) %>%
  # Convert columns back to numeric
  mutate(across(c(tmax, tmin, prcp, snow, snwd), as.numeric)) %>%
  # Continue with data transformation
  mutate(
    prcp = prcp / 10,        # Convert precipitation to mm
    tmax = tmax / 10,        # Convert tmax to °C
    tmin = tmin / 10,        # Convert tmin to °C
    tavg = (tmax + tmin) / 2 # Calculate average temperature
  ) %>%
  mutate(
    year = year(date),
    month = month(date, label = TRUE),
    day = day(date)
  )

# Calculate monthly average temperatures
monthly_avg_temp = ny_noaa_clean %>%
  filter(!is.na(tavg)) %>%
  group_by(year, month) %>%
  summarize(avg_temp = mean(tavg, na.rm = TRUE)) %>%
  ungroup()
## `summarise()` has grouped output by 'year'. You can override using the
## `.groups` argument.
# Create line plot
plot2 = monthly_avg_temp %>%
  plot_ly(
    x = ~year,
    y = ~avg_temp,
    color = ~month,
    colors = "Spectral",
    type = 'scatter',
    mode = 'lines',
    text = ~paste("Month:", month)
  ) %>%
  layout(
    title = 'Average Monthly Temperature Over Years',
    xaxis = list(title = 'Year'),
    yaxis = list(title = 'Average Temperature (°C)')
  )

# Calculate annual total precipitation
annual_precip = ny_noaa_clean %>%
  filter(!is.na(prcp)) %>%
  group_by(year) %>%
  summarize(total_prcp = sum(prcp, na.rm = TRUE)) %>%
  ungroup()

# Create bar plot
plot3 = annual_precip %>%
  plot_ly(
    x = ~year,
    y = ~total_prcp,
    type = 'bar',
    marker = list(color = 'rgba(58,200,225,.8)'),
    text = ~paste("Year:", year, "<br>Total Precipitation:", total_prcp, "mm")
  ) %>%
  layout(
    title = 'Total Annual Precipitation',
    xaxis = list(title = 'Year'),
    yaxis = list(title = 'Total Precipitation (mm)')
  )
